\section{Sample Space and Events}
\begin{itemize}
\item Set of all possible outcomes of an expansion- \textbf{\textit{sample space}},\textbf{S}
\item Subset of sample space - \textbf{\textit{event}},\textbf{E}
\end{itemize}
